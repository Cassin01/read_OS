# 03 プロセスの制御
親プロセスは子プロセスを生成し、
子プロセスの実行結果を受け取る

# プロセスのコピー　
## ``proc[]``のコピー
このとき

子プロセスの親プロセスのIDを保管する``proc.p_ppid``は親プロセスの``proc.p_pid``をコピーする

## データセグメント、PPDAのコピー
* 子プロセスの``user.u_procpは``自身の``user``構造体に対応した子プロセスの``proc[]``エントリを持つ

* テキストセグメントは親プロセスの同じ``text[]``エントリを指すため
コピーされない

# お互いどのように参照するか
## 子プロセスが親プロセスを知る
``proc.p_ppid``に親プロセスのIDが設定されるのでそこを見れば良い
## 親プロセスが自分の子プロセスを知る
(子プロセスの)``proc[]``を全て調べ
自身の親プロセスのIDを示す``proc.p_ppid``が自身を指しているプロセスを見つける必要がある

# forkシステムコール
システムコールの処理はカーネルプロセスにより行われる

forkシステムコールからは、親プロセスと子プロセス用に2回返される　

```c
// fork() (sys/ken.c)
// 親プロセスの処理
fork() {
  register struct proc *p1, *p2;

  p1 = u.u_procp; // 実行プロセス(親プロセスのproc構造体をp1に格納)

  // proc[]の空きエントリを探す
  for(p2 = &proc[0]; p2 , &proc[NPROC]; p2++) {
    if (p2 -> p_start = NULL) {
      goto found;
    }
  }

  // EAGAIN: forkシステムコールで、proc[]に空きエントリが見つからなかった
  u.u_error = EAGAIN;

  goto out

found:
 // newproc(): 新しいプロセスを生成 (親プロセス: 0, 子プロセス: 1)
 if(newproc()) {
    // p_pid(親プロセスのID)をユーザプロセスのr0に格納
    // これが子プロセスへのforkシステムコールからの戻り値
    u.u_ar0[R0] = p1->p_pid;

    // プロセスの実行時間情報を0に初期化し直す
    u.u_cstime[0] = 0;
    u.u_cstime[1] = 0;
    u.u_stime = 0;
    u.u_cutime[0] = 0;
    u.u_cutime[1] = 0;
    u.u_utime = 0;
    return;
  }

  // 子プロセスのproc.p_pidを親プロセスのr0に格納
  // p_pid: プロセスのID
  // これがforkシステムコールの戻り値となる
  u.u_ar0[R0] = p2->p_pid;

out:
  // 親プロセスのプログラムカウンタを1命令分進める
  u.u_ar0[R7] =+ 2;
}
```

```assembly
/ Cライブラリのfork() (source/s4/fork.s)
/ PDPアセンブリではバックスラッシュはコメント

.globl  _fork, cerror, _par_uid

_fork:
  mov   r5,-(sp)
  mov   sp,r5
  sys   fork
    / brはジャンプ命令
    / 1fは前方の１ラベル(1:)にジャンプすることを表す
    br  1f
  bec   2f
  jmp   cerror

1:
  / 親プロセスのproc.p_pidを_par_uidにコピー
  / Cライブラリのfork()を実行したユーザプログラムはこの値を見ることで親プロセスのIDを取得できる
  mov   r0,_par_uid

  / r0をクリア
  clr   r0

2:
  mov   (sp)+,r5

  / rts命令でCライブラリのfork()呼び出し元へ戻りr0に格納された0が返る
  rts   pc

.bss
_par_uid: .=.+2
```

# newproc()
プロセスの生成をする関数

```c
// newproc() (ken/slp.c)
newproc() {
  int a1, a2;
  struct proc *p, *up;
  register struct proc *rpp;
  register *rip, n;

  p = NULL;
retry:
  /* IDの生成 */

  // プロセスを生成するごとにmpidをインクリメントする
  // mpidが子プロセスに付与されるIDになる
  mpid++;

  // mpidはint型であるからインクリメントしていくと最上位の符号ビットが立ち
  // 負の値となるのでこのときmpidを0に戻してリトライする
  if(mpid < 0) {
    mpid = 0;
    goto retry;
  }
  /* proc[]の空きエントリを探す */
  for(rpp = &proc[0]; rpp < &proc[NPROC]; rpp++) {
    if(rpp->p_stat == NULL && p==NULL)
      p = rpp;
    // 新たに割り当てようとしているIDと同じIDを持っているプロセスがあればリトライ
    if (rpp->p_pid==mpid)
      goto retry;
  }
  // 空いているproc[]がない場合
  if ((rpp = p)==NULL)
    panic("no procs");

  // 子プロセスの
  /* proc[]エントリの生成 */
  // 実行プロセス(親プロセス)のproc[]エントリを取得
  // ちなみにu_procpはuser構造体に対応したproc[]エントリをさす
  rip = u.u_procp;
  up = rip;
  // 実行可能状態
  rpp->p_stat  = SRUN;

  // メモリ中に存在する
  rpp->p_flag  = SLOAD;

  // IDを割り当て
  rpp->p_uid   = rip->p_uid;
  rpp->p_ttyp  = rip->p_ttyp;
  rpp->p_nice  = rip->p_nice;
  rpp->p_textp = rip->p_textp;
  rpp->p_pid   = mpid;
  親プロセスが実行プロセス
  rpp->p_ppid  = rip->p_pid;
  実行時間が0
  rpp->p_time  = 0;

  /* 参照カウンタをインクリメント */
  // 参照カウンタとは
  // オブジェクトを参照するポインタの数を数え、
  // 参照するポインタの数がゼロになったら解放する方法。
  // 親プロセスと同じtext[]エントリの参照カウンタをインクリメントする
  for(rip = &u.u_ofile[0]; rip < &u.u_ofile[NOFILE];)
    if((rpp = *rip++) != NULL)
      rpp->f_count++;
  if((rpp=up->p_textp) != NULL) {
    rpp->x_count++;
    rpp->x_ccount++;
  }
  // カレントディレクトリのinode[]エントリの参照カウンタをインクリメント
  u.u_cdir->i_count++;

  /* データセグメントのコピー */
  r5, r6 をu.u_rsavに待機
  savu(u.u_rsav);

  // 一時的に親プロセスのuser.u_procpから子プロセスのproc[]エントリをさす
  rpp = p;

  // 子プロセスのu.u_procpから子プロセスのproc[]エントリを指すようになる
  u.u_procp = rpp;

  // ripに親プロセスのproc[]エントリを退避
  rip = up;

  n = rip->p_size;
  a1 = rip->p_addr;
  rpp->p_size = n;
  a2 = malloc(coremap, n);
  if(a2 == NULL) {
    // SIDL: プロセス生成中(データが変化してしまうのを防ぐ)
    // 親プロセスは上のスワップ処理が完了するかで休眠状態に入る
    rip->p_stat = SIDL;
    rpp->p_addr = a1;
    savu(u.u_ssav);
    xswap(rpp, 0, 0);
    rpp->p_flag =| SSWAP;
    rip->p_stat = SRUN;
  } else {
    rpp->p_addr = a2;
    while(n--)
      copyseg(a1++, a2++);
  }
  u.u_procp = rip;
  // 親プロセスの場合0が返る
  return(0);
}
```

## スワップ
メモリが不足しているときに一時的に不必要なプロセスを退避させ
退避させたプロセスが使っていたメモリで実行すること

```c
// panic() (ken/prf.c)
panic(s)
char *s;
{
  panicstr = s;
  update();
  printf("panic: %s\n", s);
  for(;;)
    // idle()を無限に呼び出し処理の継続を止める

```

```assembly
.globl _idle
_idle:
  / プロセッサ優先度0(かなり高い方)
  mov PS,-(sp)
  bic $340,PS
  wait
  mov (sp)+,PS
  rts ps
```

# sleep() (ken/slp.c)
実行中のプロセスを休眠状態(SSLEEP, SWAIT)にする
switch()を実行して実行プロセスを切り替える

```c
// waiting_channel 待っていいる資源のアドレス
// pri 実行優先度
sleep(chan, pri) {
  register *rp, s;

  // PSWを退避
  s = PS->integ;

  // proc[]エントリ格納
  rp = u.u_procp;

  // pri
  // 0以上: SWAIT状態になる。シグナル処理を行う
  // 0未満: SSLEEP状態になる。シグナル処理は行わない
  if(pri >= 0) {
    if(issig())
      goto psig;
    spl6();
    // プロセスの状態を実行可能状態へと変更するwakeup()は
    // proc.p_wchanを12調べ引数と一致するプロセスを(SRUN)にする
    rp->p_wchan = chan;
    rp->p_stat = SWAIT;
    rp->p_pri = pri;
    spl0();
    if(runin != 0) {
      runin = 0;
      wakeup(&runin);
    }
    swtch();
    if(issig())
      goto psig;
  } else {
    spl6();
    rp->p_wchan = chan;
    rp->p_stat = SSLEEP;
    rp->p_pri = pri;
    spl0();
    swtch();
  }

  // 中断されていたプロセスが再実行される
  // 退避していたPSWを戻す
  PS->integ = s;
  return;
psig:
  aretu(u.u_qsav);
}
```

# swtch()
実行プロセスの切り替えを行う

# やること
* 実行可能状態なプロセスのうち、もっとも実行優先度の高いプロセスを選択する
* 選択されたプロセスのuser構造体をuで参照できるようにカーネルAPRを変更
* 選択されたプロセスのuser構造体に保存されているr5, r6の値を復帰
* 選択されたプロセスのuser構造体に保存されているユーザAPRの値を復帰

```c
// switch() (ken/slp.c)
swtch() {
  static struct proc *p;
  register i, n;
  register struct proc *rp;

  if(p == NULL)
    p = &proc[0];

  savu(u.u_rsav);
  retu(proc[0].p_addr);

loop:
 runrun = 0;
 rp = p;
 p = NULL;
 n = 128;

 i = NPROC;
  do {
    rp++;
    if(rp >= &proc[NPROC])
        rp = &proc[0];
    if(rp->p_stat==SRUN && (rp->p_flag&SLOAD)!=0) {
        if(rp->p_pri < n) {
            p = rp;
            n = rp->p_pri;
        }
    }
  } while(--i);

  if(p == NULL) {
    p = rp;
    idle();
    goto loop;
  }
  rp = p;
  curpri = n;

  retu(rp->p_addr);
  sureg();

  if(rp->p_flag&SSWAP) {
    rp->p_flag =& ~SSWAP;
    aretu(u.u_ssav);
  }

  return(1);
}
```
